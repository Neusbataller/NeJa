package dao;

import db.DBConnection;
import model.Alumno;

import java.sql.*;
import java.util.ArrayList;
import java.util.List;

/**
 * Implementaci√≥n del DAO usando JDBC.
 *
 * Aqu√≠ va el "c√≥mo se hace":
 * - SQL
 * - PreparedStatement
 * - ResultSet
 */
public class AlumnoDAOImpl implements AlumnoDAO {

    @Override
    public boolean insertar(Alumno alumno) {

        // SQL con ? porque usaremos PreparedStatement (m√°s seguro y f√°cil)
        String sql = "INSERT INTO alumnos (nombre, edad) VALUES (?, ?)";

        // try-with-resources:
        // Se encarga de cerrar autom√°ticamente la conexi√≥n y el statement.
        try (Connection conn = DBConnection.getConnection();
             PreparedStatement ps = conn.prepareStatement(sql)) {

            // 1¬∫ ? corresponde a nombre
            ps.setString(1, alumno.getNombre());
            // 2¬∫ ? corresponde a edad
            ps.setInt(2, alumno.getEdad());

            // executeUpdate se usa en INSERT/UPDATE/DELETE
            // devuelve n√∫mero de filas afectadas
            int filas = ps.executeUpdate();

            return filas > 0; // si insert√≥ 1 fila, true

        } catch (SQLException e) {
            System.out.println("Error al insertar alumno: " + e.getMessage());
            return false;
        }
    }

    @Override
    public List<Alumno> listar() {

        String sql = "SELECT id, nombre, edad FROM alumnos";
        List<Alumno> lista = new ArrayList<>();

        try (Connection conn = DBConnection.getConnection();
             PreparedStatement ps = conn.prepareStatement(sql);
             ResultSet rs = ps.executeQuery()) {

            // ResultSet = "tabla" con resultados
            while (rs.next()) {
                Alumno a = new Alumno();

                // Sacamos cada columna por su nombre
                a.setId(rs.getInt("id"));
                a.setNombre(rs.getString("nombre"));
                a.setEdad(rs.getInt("edad"));

                lista.add(a);
            }

        } catch (SQLException e) {
            System.out.println("Error al listar alumnos: " + e.getMessage());
        }

        return lista;
    }

    @Override
    public Alumno buscarPorId(int id) {

        String sql = "SELECT id, nombre, edad FROM alumnos WHERE id = ?";
        Alumno alumno = null;

        try (Connection conn = DBConnection.getConnection();
             PreparedStatement ps = conn.prepareStatement(sql)) {

            ps.setInt(1, id);

            try (ResultSet rs = ps.executeQuery()) {
                // Si existe, rs.next() ser√° true
                if (rs.next()) {
                    alumno = new Alumno();
                    alumno.setId(rs.getInt("id"));
                    alumno.setNombre(rs.getString("nombre"));
                    alumno.setEdad(rs.getInt("edad"));
                }
            }

        } catch (SQLException e) {
            System.out.println("Error al buscar alumno por id: " + e.getMessage());
        }

        return alumno;
    }

    @Override
    public boolean actualizar(Alumno alumno) {

        String sql = "UPDATE alumnos SET nombre = ?, edad = ? WHERE id = ?";

        try (Connection conn = DBConnection.getConnection();
             PreparedStatement ps = conn.prepareStatement(sql)) {

            ps.setString(1, alumno.getNombre());
            ps.setInt(2, alumno.getEdad());
            ps.setInt(3, alumno.getId());

            int filas = ps.executeUpdate();
            return filas > 0;

        } catch (SQLException e) {
            System.out.println("Error al actualizar alumno: " + e.getMessage());
            return false;
        }
    }

    @Override
    public boolean eliminar(int id) {

        String sql = "DELETE FROM alumnos WHERE id = ?";

        try (Connection conn = DBConnection.getConnection();
             PreparedStatement ps = conn.prepareStatement(sql)) {

            ps.setInt(1, id);

            int filas = ps.executeUpdate();
            return filas > 0;

        } catch (SQLException e) {
            System.out.println("Error al eliminar alumno: " + e.getMessage());
            return false;
        }
    }
}
